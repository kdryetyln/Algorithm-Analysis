def fibo_loop(n):
    if(n<2):
        return n
    else:
        a=0
        b=1
    while(n>0):
        c=a+b
        a=b
        b=c
        n=n-1
    return c

def fibo_rec(n):
    if(n<2):
        return n
    else:
        return fibo_rec(n-1)+fibo_rec(n-2)

import matplotlib.pyplot as plt
for n in range(40):
    time1=int(round(time.time()))
    fibo_rec(n)
    time2=int(round(time.time()))
    print("recursive ve np olan :",n," iin geen sre : ",time2-time1,"saniye")
    time3=int(round(time.time()))
    fibo_loop(n)
    time4=int(round(time.time()))
    print("-----------------------------------------------------------")
    print("loop p(linear) olan : ",n," iin geen sre : ",time4-time3,"saniye")
    plt.plot([n,time2-time1])
    plt.plot([n,time4-time3])
    plt.xlabel("Orange : np values\n Blue : loop values")
    plt.show()
